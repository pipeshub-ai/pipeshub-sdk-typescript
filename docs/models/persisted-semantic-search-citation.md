# PersistedSemanticSearchCitation

Populated citation document referenced from a persisted search.
The controller strips `__v` via `populate({ select: '-__v' })`.


## Example Usage

```typescript
import { PersistedSemanticSearchCitation } from "@pipeshub-ai/sdk/models";

let value: PersistedSemanticSearchCitation = {
  id: "<value>",
  content: "<value>",
  chunkIndex: 153298,
  citationType: "<value>",
  metadata: {
    orgId: "<id>",
    mimeType: "<value>",
    recordId: "<id>",
    recordName: "<value>",
    origin: "<value>",
  },
  createdAt: new Date("2024-07-18T15:25:34.738Z"),
  updatedAt: new Date("2024-12-18T20:26:13.397Z"),
};
```
